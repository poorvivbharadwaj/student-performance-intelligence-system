import streamlit as st
import pandas as pd
import numpy as np
from sklearn.linear_model import LinearRegression
import plotly.express as px
from reportlab.platypus import SimpleDocTemplate, Paragraph, Spacer
from reportlab.lib.styles import getSampleStyleSheet
from io import StringIO

# ----------------------------
# PAGE CONFIG
# ----------------------------
st.set_page_config(page_title="Student Intelligence System", layout="wide")

st.title("🎓 Smart Student Performance Intelligence System")

# ----------------------------
# SAMPLE DATA

# ----------------------------
# SIDEBAR
# ----------------------------
st.sidebar.title("📊 Controls")

st.sidebar.download_button(
    "📥 Download Sample Dataset",
    data=get_sample_csv(),
    file_name="students_sample.csv",
    mime="text/csv"
)

uploaded_file = st.sidebar.file_uploader("📂 Upload CSV", type=["csv"])

def load_sample():
    return pd.read_csv(StringIO(get_sample_csv()))

if uploaded_file:
    df = pd.read_csv(uploaded_file)
    st.sidebar.success("Custom dataset uploaded")
else:
    df = load_sample()
    st.sidebar.info("Using sample dataset")

# ----------------------------
# CLEAN DATA
# ----------------------------
df.columns = df.columns.str.strip().str.lower().str.replace(" ", "_")
df = df.loc[:, ~df.columns.duplicated()]

numeric_cols = df.select_dtypes(include=np.number).columns.tolist()
name_col = [col for col in df.columns if "name" in col][0]

df = df.fillna(df.mean(numeric_only=True))

# ----------------------------
# FEATURES
# ----------------------------
target_cols = ["attendance", "study_hours", "previous_score"]
subject_cols = [col for col in numeric_cols if col not in target_cols]

df["Average"] = df[subject_cols].mean(axis=1)
df["Trend"] = df["Average"] - df.get("previous_score", 0)

def get_risk(row):
    if row["Average"] < 40 or row.get("attendance", 100) < 60:
        return "High"
    elif row["Average"] < 60:
        return "Moderate"
    return "Low"

df["Risk"] = df.apply(get_risk, axis=1)

# ----------------------------
# MODEL (Linear Regression)
# ----------------------------
features = subject_cols.copy()
extra_cols = ["attendance", "study_hours", "assignments",
              "sports", "participation", "sleep_hours",
              "screen_time", "stress_level"]

for col in extra_cols:
    if col in df.columns:
        features.append(col)

features = list(dict.fromkeys(features))

model = LinearRegression()
model.fit(df[features], df["Average"])

# ----------------------------
# TABS UI
# ----------------------------
tab1, tab2, tab3, tab4 = st.tabs([
    "📊 Overview",
    "👤 Student Analysis",
    "📈 Insights",
    "📂 Dataset"
])

# ----------------------------
# OVERVIEW
# ----------------------------
with tab1:
    st.subheader("📊 Class Dashboard")

    col1, col2, col3, col4 = st.columns(4)
    col1.metric("Avg Score", round(df["Average"].mean(), 2))
    col2.metric("Top Score", round(df["Average"].max(), 2))
    col3.metric("Lowest Score", round(df["Average"].min(), 2))
    col4.metric("High Risk", len(df[df["Risk"] == "High"]))

    st.markdown("---")

    colA, colB = st.columns(2)
    with colA:
        st.plotly_chart(px.histogram(df, x="Average"), use_container_width=True)
    with colB:
        st.plotly_chart(px.pie(df, names="Risk"), use_container_width=True)

    top = df.loc[df["Average"].idxmax()]
    st.success(f"🏆 Top Performer: {top[name_col]}")

# ----------------------------
# STUDENT ANALYSIS
# ----------------------------
with tab2:
    st.subheader("👤 Student Analysis")

    student = st.selectbox("Select Student", df[name_col])
    row = df[df[name_col] == student].iloc[0]

    col1, col2, col3 = st.columns(3)
    col1.metric("Average", round(row["Average"], 2))
    col2.metric("Trend", round(row["Trend"], 2))
    col3.metric("Attendance", row.get("attendance", "N/A"))

    st.plotly_chart(
        px.bar(
            x=subject_cols,
            y=[row[col] for col in subject_cols]
        ),
        use_container_width=True
    )

    st.subheader("Progress")
    for sub in subject_cols:
        st.write(sub)
        st.progress(int(row[sub]))

    weak = min(subject_cols, key=lambda x: row[x])
    st.error(f"Weak Subject: {weak}")

    pred = model.predict([[row[col] for col in features]])
    st.success(f"Predicted Score: {round(pred[0],2)}")

    if st.button("Generate AI Insight"):
        if row["Risk"] == "High":
            st.write("High risk. Improve study hours and attendance.")
        elif row["Risk"] == "Moderate":
            st.write("Moderate performance. Improve consistency.")
        else:
            st.write("Good performance.")

    q = st.text_input("Ask AI")
    if q:
        st.write("Focus on consistency and practice.")

# ----------------------------
# INSIGHTS
# ----------------------------
with tab3:
    st.subheader("📈 Class Insights")

    corr = df.select_dtypes(include=np.number).corr()
    st.plotly_chart(px.imshow(corr, text_auto=True), use_container_width=True)

    st.plotly_chart(
        px.line(df, x=name_col, y=subject_cols, markers=True),
        use_container_width=True
    )

    risk_filter = st.selectbox("Filter by Risk", ["All", "High", "Moderate", "Low"])

    if risk_filter != "All":
        filtered = df[df["Risk"] == risk_filter]
    else:
        filtered = df

    st.dataframe(filtered)

# ----------------------------
# DATASET
# ----------------------------
with tab4:
    st.subheader("📂 Dataset View")

    st.write(f"Rows: {df.shape[0]} | Columns: {df.shape[1]}")
    st.dataframe(df)

    st.subheader("Statistics")
    st.dataframe(df.describe())

# PDF REPORT
# ----------------------------
def generate_pdf(row):
    doc = SimpleDocTemplate("report.pdf")
    styles = getSampleStyleSheet()

    content = [
        Paragraph("Student Report", styles["Title"]),
        Spacer(1, 10),
        Paragraph(f"Name: {row['name']}", styles["Normal"]),
        Paragraph(f"Average: {row['Average']}", styles["Normal"]),
        Paragraph(f"Risk: {row['Risk']}", styles["Normal"]),
    ]

    doc.build(content)
    return "report.pdf"

if "row" in locals():
    if st.button("Download Report"):
        file = generate_pdf(row)
        with open(file, "rb") as f:
            st.download_button("Download PDF", f, "report.pdf")
